40float cossenoSimilaridade_TF(int *a, int *b, int size) {
41 float produtoInt = produtoInterno_TF(a, b, size);
42 float cossenoSim = 0.0f;
43 cossenoSim = produtoInt / (normalCossenoSimilaridade_TF(a, size) * normalCossenoSimilaridade_TF(b, size));
44 return cossenoSim;
45}
